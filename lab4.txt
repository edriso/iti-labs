CREATE TABLE users (
	user_id int AUTO_INCREMENT,
	name varchar(30) NOT NULL,
	email varchar(40),
	gender ENUM('F', 'M'),
	send_emails ENUM('yes', 'no'),
	PRIMARY KEY(user_id)
);
